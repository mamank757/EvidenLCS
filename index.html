// 1. Definisikan variabel utama di awal agar tidak error const tbody = document.getElementById('table-body'); let jumlahBaris = parseInt(localStorage.getItem('jumlahBaris')) || 2; // 2. Fungsi mendapatkan tanggal hari ini function getTanggalHariIni() { const tgl = new Date(); const yyyy = tgl.getFullYear(); const mm = String(tgl.getMonth() + 1).padStart(2, '0'); const dd = String(tgl.getDate()).padStart(2, '0'); return `${yyyy}-${mm}-${dd}`; } // 3. Fungsi membuat baris tabel function renderBaris(i) { const tr = document.createElement('tr'); tr.innerHTML = ` ${i}
FOTO
LIKE
FOTO
COMMENT
FOTO
SHARE
`; return tr; } // 4. Fungsi memuat data (dipanggil saat body onload) function muatDataLokal() { tbody.innerHTML = ""; for (let i = 1; i <= jumlahBaris; i++) { tbody.appendChild(renderBaris(i)); // Muat Tanggal const inputTgl = document.getElementById(`tgl_${i}`); const tglSimpan = localStorage.getItem(`tgl_${i}`); inputTgl.value = tglSimpan ? tglSimpan : getTanggalHariIni(); // Muat Judul const judulSimpan = localStorage.getItem(`judul_${i}`); if(judulSimpan) document.getElementById(`judul_${i}`).value = judulSimpan; // Muat Gambar ['like', 'comen', 'share'].forEach(type => { const idImg = `img_${type}_${i}`; const savedImg = localStorage.getItem(idImg); if(savedImg) { const imgEl = document.getElementById(idImg); if(imgEl) { imgEl.src = savedImg; imgEl.style.display = 'block'; } } }); } // Muat Identitas ['id_periode', 'id_nama', 'id_nip', 'id_wilayah'].forEach(id => { const val = localStorage.getItem(id); if(val) document.getElementById(id).value = val; }); } // 5. Fungsi tambah baris function tambahBaris() { jumlahBaris++; localStorage.setItem('jumlahBaris', jumlahBaris); const tr = renderBaris(jumlahBaris); tbody.appendChild(tr); const tglBaru = document.getElementById(`tgl_${jumlahBaris}`); tglBaru.value = getTanggalHariIni(); localStorage.setItem(`tgl_${jumlahBaris}`, tglBaru.value); } // 6. Fungsi pendukung lainnya (PENTING!) function simpanTeks(id) { const el = document.getElementById(id); if(el) localStorage.setItem(id, el.value); } function prosesGambar(input, imgId) { if (input.files && input.files[0]) { const reader = new FileReader(); reader.onload = function(e) { const img = document.getElementById(imgId); img.src = e.target.result; img.style.display = 'block'; try { localStorage.setItem(imgId, e.target.result); } catch(err) { alert("Memori penuh! Coba gunakan foto yang lebih kecil."); } } reader.readAsDataURL(input.files[0]); } } function formatNIP(input) { input.value = input.value.replace(/\D/g, '').substr(0, 18); } function simpanPDFAndroid() { const element = document.getElementById('capture-area'); const toolbar = document.getElementById('toolbar'); const labels = document.querySelectorAll('.label-eviden'); const nama = document.getElementById('id_nama').value || "PPL"; toolbar.style.display = 'none'; labels.forEach(el => el.style.visibility = 'hidden'); const opt = { margin: [8, 5, 8, 5], filename: `LCS_${nama}.pdf`, image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, useCORS: true }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'portrait' } }; html2pdf().set(opt).from(element).save().then(() => { toolbar.style.display = 'flex'; labels.forEach(el => el.style.visibility = 'visible'); }); } function tampilkanPopup() { document.getElementById('popup-reset').style.display = 'flex'; } function tutupPopup() { document.getElementById('popup-reset').style.display = 'none'; } function konfirmasiReset() { localStorage.clear(); location.reload(); }
